1%'invpd' Inverse of a symmetric positive definite matrix using Cholesky factorization 
 2%Ainv = invpd(A) computes the inverse of a symmetric positive 
 3%definite matrix A using its Cholesky factor H.
 4%inv(A) = inv(H)inv(H)'. 
 5%input  : Matrix A
 6%output : Ainv, err (=1 if error, and 0 if no error)
 7%대칭, 양정부호 행렬의 역행렬을 LU분해보다 빠르고 정확하게 계산해 줍니다.
 8function [Ainv,err] = invpd(A)
 9    [m,n] = size(A);
10    err = 0;
11    if m~=n
12            disp('matrix A  is not square');
13            err = 1;
14            Ainv = [];
15            return
16    end
17    
18    [H,err] = cholmod(A);
19    if err == 1  % = 1 if error
20          %  disp('matrix A  is not positive definite');
21            Ainv = [];
22    else 
23            Hinv = invuptr(H);
24            Ainv = Hinv*Hinv';
25    end
26    Ainv = (Ainv+Ainv')/2;
27end
