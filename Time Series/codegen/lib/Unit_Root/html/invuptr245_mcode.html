 3%T = invuptr(T) computes the inverse of a nonsingular upper triangular
 4%matrix T.  The output matrix T contains the inverse of T. 
 5%This program implements Algorithm 4.2.2 of the book.
 6%Input  : Matrix T 
 7%output : Matrix T
 8
 9    [m,n] = size(T);
10        if m~=n
11            disp('matrix T  is not square')

13        end;
14        s = eye(n,n);
15    for k = n:-1:1
16          if ( T(k,k) == 0)   
17            disp('matrix T is singular')

19          end;
20      T(k,k) = 1/T(k,k);
21      for i = k-1 :-1 :1
22        sum = 0;

24        T(i,k) = -sum/T(i,i);
