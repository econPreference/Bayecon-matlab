1% function [R, indef, E, err] = cholmod(A)
 2function [R, err] = cholmod(A)
 3% CHOLMOD Modified Cholesky factorization
 4%  R = cholmod(A) returns the upper Cholesky factor of A (same as CHOL)
 5%  if A is (sufficiently) positive definite, and otherwise returns a 
 6%  modified factor R with diagonal enries >= sqrt(delta) and
 7%  offdiagonal entries <= beta in absolute value,
 8%  where delta and beta are defined in terms of size of diagonal and
 9%  offdiagonal entries of A and the machine precision; see below.
10%  The idea is to ensure that E = A - R'*R is reasonably small if A is 
11%  not too far from being indefinite.  If A is sparse, so is R.
12%  The output parameter indef is set to 0 if A is sufficiently positive
13%  definite and to 1 if the factorization is modified.
14%
15%  The point of modified Cholesky is to avoid computing eigenvalues, 
16%  but for dense matrices, MODCHOL takes longer than calling the built-in 
17%  EIG, because of the cost of interpreting the code, even though it
18%  only has one loop and uses vector operations.  
19
20%  reference: Nocedal and Wright, Algorithm 3.4 and subsequent discussion
21%  (not Algorithm 3.5, which is more complicated)
22%  original algorithm is due to Gill and Murray, 1974
23%  written by M. Overton, overton@cs.nyu.edu, last modified Feb 2005
24
25%  convenient to work with A = LDL' where D is diagonal, L is unit
26%  lower triangular, and so R = (LD^(1/2))'
27%
28
29coder.extrinsic('speye')
30
31err = 0;
32if sum(sum(abs(A-A'))) > 0
33    err = 1;
34    error('A is not symmetric')
35end
36
37% set parameters governing bounds on L and D (eps is machine epsilon)
38
39n = length(A);
40diagA = diag(A);
41gamma = max(abs(diagA));             % max diagonal entry
42xi = max(max(abs(A - diag(diagA))));  % max off-diagonal entry
43delta = eps*(max([gamma+xi, 1]));
44beta = sqrt(max([gamma, xi/n, eps]));
45indef = 0;
46
47% initialize d and L
48
49d = zeros(n,1);
50if issparse(A)
51    L = speye(n);  % sparse identity
52else
53    L = eye(n);    
54end
55
56% there are no inner for loops, everything implemented with
57% vector operations for a reasonable level of efficiency
58
59for j = 1:n
60    K = 1:j-1;  % column index: all columns to left of diagonal
61                % d(K) doesn't work in case K is empty
62    djtemp = A(j,j) - L(j,K)*(d(K,1).*L(j,K)');    % C(j,j) in book
63    if j < n
64        I = j+1:n;  % row index: all rows below diagonal
65        Ccol = A(I,j) - L(I,K)*(d(K,1).*L(j,K)');  % C(I,j) in book
66        theta = max(abs(Ccol));
67        % guarantees d(j) not too small and L(I,j) not too big
68        % in sufficiently positive definite case, d(j) = djtemp
69        d(j) = max([abs(djtemp), (theta/beta)^2, delta]);
70        L(I,j) = Ccol/d(j);
71    else
72        d(j) = max([abs(djtemp), delta]);
73    end
74    if d(j) > djtemp  % A was not sufficiently positive definite
75        indef = 1;
76    end
77end
78% convert to usual output format: replace L by L*sqrt(D) and transpose
79for j=1:n
80    L(:,j) = L(:,j)*sqrt(d(j));   % L = L*diag(sqrt(d)) bad in sparse case
81end;
82R = L';
83if nargout == 3
84    E = A - R'*R;
85end
