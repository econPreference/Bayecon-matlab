1%   to compute the inverted gamma density on a grid
 2%   beta is also a vector 
 3%   outputs the log of the density
 4function [retf] = lnpdfig(sig2,a,b)
 5
 6if rows(sig2) > rows(a)
 7   alpha = a*ones(rows(sig2),1);
 8   beta = b*ones(rows(sig2),1);
 9else
10   alpha = a;
11   beta = b;
12end  
13% sig2가 벡터인 경우를 대비
14% 이 경우, 각각의 원소에 대해 우도를 계산해줌
15
16   c = alpha.*log(beta) - gammaln(alpha);
17   z = c - (alpha+1).*log(sig2) - beta./sig2;
18
19 
20retf = z;
21end
